    public class UserDetails extends AppCompatActivity{
        @Override
        protected void onCreate(@Nullable Bundle savedInstanceState) {
            super.onCreate(savedInstanceState);
            setContentView(R.layout.user_details);

            Button submitData = findViewById(R.id.submitData);
            EditText txtName = findViewById(R.id.txtName);
            TextView messageshow1 = findViewById(R.id.messageshow1);
            TextView messageshow2 = findViewById(R.id.messageshow2);
            TextView messageshow3 = findViewById(R.id.messageshow3);
            RadioButton male = findViewById(R.id.male);
            RadioButton female = findViewById(R.id.female);
            CheckBox chkbox1 = findViewById(R.id.chkbox1);
            CheckBox chkbox2 = findViewById(R.id.chkbox2);
            CheckBox chkbox3 = findViewById(R.id.chkbox3);
            CheckBox chkbox4 = findViewById(R.id.chkbox4);


            submitData.setOnClickListener(new View.OnClickListener() {
                @Override
                public void onClick(View v) {
                    String name = txtName.getText().toString();
                    String message1 = "User Details: \nName : " + name;
                    messageshow1.setText(message1);
                    txtName.setText("");

                    if (male.isChecked())
                    {
                        messageshow2.setText("Gender: " + male.getText().toString());

                    }
                    else if (female.isChecked())
                    {
                        messageshow2.setText("Gender: " + female.getText().toString());
                    }
                    else{
                        messageshow2.setText("Gender: Not selected");
                    }

                    String message3 = "Selected Subjects: \n";
                    if(chkbox1.isChecked()){
                        message3 = message3 + "\t\t" +chkbox1.getText().toString() + "\n";
                    }
                    if (chkbox2.isChecked())
                    {
                        message3 = message3 + "\t\t" +chkbox2.getText().toString() + "\n";
                    }
                    if(chkbox3.isChecked()){
                        message3 = message3 + "\t\t" +chkbox3.getText().toString() + "\n";
                    }
                    if (chkbox4.isChecked())
                    {
                        message3 = message3 + "\t\t" +chkbox4.getText().toString() + "\n";
                    }
                    messageshow3.setText(message3);
                }
            });
        }
    }
